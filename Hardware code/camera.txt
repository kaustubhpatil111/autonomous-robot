#include "esp_camera.h"
#include <WiFi.h>
#include <WebServer.h>

// Wi-Fi credentials
const char* ssid = "JioFi3_230593";
const char* password = "kpKP2005@";

// AI Thinker pin definitions (same as provided)
#define PWDN_GPIO_NUM    32
#define RESET_GPIO_NUM   -1
#define XCLK_GPIO_NUM     0
#define SIOD_GPIO_NUM    26
#define SIOC_GPIO_NUM    27
#define Y9_GPIO_NUM      35
#define Y8_GPIO_NUM      34
#define Y7_GPIO_NUM      39
#define Y6_GPIO_NUM      36
#define Y5_GPIO_NUM      21
#define Y4_GPIO_NUM      19
#define Y3_GPIO_NUM      18
#define Y2_GPIO_NUM       5
#define VSYNC_GPIO_NUM   25
#define HREF_GPIO_NUM    23
#define PCLK_GPIO_NUM    22

WebServer server(80);

// Stream handler: sends a multipart JPEG stream
void handleStream() {
  WiFiClient client = server.client();
  if (!client) return;

  client.println("HTTP/1.1 200 OK");
  client.println("Content-Type: multipart/x-mixed-replace; boundary=frame");
  client.println();

  // Continuously capture and send frames
  while (client.connected()) {
    camera_fb_t *fb = esp_camera_fb_get();
    if (!fb) break; // Stop if frame capture fails

    // Send frame header and JPEG buffer
    client.printf("--frame\r\nContent-Type: image/jpeg\r\nContent-Length: %u\r\n\r\n", fb->len);
    client.write(fb->buf, fb->len);
    client.print("\r\n");

    esp_camera_fb_return(fb);
    // Short delay (tunable for desired FPS; 40ms ~ 25 FPS)
    // Reduce or remove this delay if the network can handle higher FPS.
    delay(40);
  }
  client.stop();
}

void setup() {
  Serial.begin(115200);

  // Camera configuration
  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer   = LEDC_TIMER_0;
  config.pin_d0       = Y2_GPIO_NUM;
  config.pin_d1       = Y3_GPIO_NUM;
  config.pin_d2       = Y4_GPIO_NUM;
  config.pin_d3       = Y5_GPIO_NUM;
  config.pin_d4       = Y6_GPIO_NUM;
  config.pin_d5       = Y7_GPIO_NUM;
  config.pin_d6       = Y8_GPIO_NUM;
  config.pin_d7       = Y9_GPIO_NUM;
  config.pin_xclk     = XCLK_GPIO_NUM;
  config.pin_pclk     = PCLK_GPIO_NUM;
  config.pin_vsync    = VSYNC_GPIO_NUM;
  config.pin_href     = HREF_GPIO_NUM;
  config.pin_sscb_sda = SIOD_GPIO_NUM;
  config.pin_sscb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn     = PWDN_GPIO_NUM;
  config.pin_reset    = RESET_GPIO_NUM;
  config.xclk_freq_hz = 20000000;
  config.pixel_format = PIXFORMAT_JPEG;

  // === OPTIMIZED SETTINGS ===
  // Increase frame size for better detail (e.g. SVGA=800x600 or VGA=640x480),
  // but be mindful of latency: larger frames take longer to send. 
  // For real-time streaming, SVGA or VGA often strikes a good balance:contentReference[oaicite:0]{index=0}.
  config.frame_size   = FRAMESIZE_SVGA;
  config.jpeg_quality = 12;   // 0-63 (lower = higher quality). 12 is moderate quality.
  // Use 2 frame buffers if PSRAM is available; this can slightly improve FPS.
  // If you encounter blank streams or timeouts, try fb_count = 1 to reduce memory pressure:contentReference[oaicite:1]{index=1}.
  config.fb_count     = 2;
  config.grab_mode    = CAMERA_GRAB_LATEST; // Always grab the most recent frame (lowest latency)

  if (esp_camera_init(&config) != ESP_OK) {
    Serial.println("Camera init failed");
    return;
  }

  // Optional: further tune sensor parameters (contrast, brightness, etc).
  sensor_t *s = esp_camera_sensor_get();
  s->set_framesize(s, FRAMESIZE_SVGA);
  s->set_quality(s, 12);
  s->set_brightness(s, 1);
  s->set_contrast(s, 1);
  // Other sensor settings (exposure, AWB, etc.) can be tweaked as needed.

  // Connect to Wi-Fi
  Serial.print("Connecting to Wi-Fi");
  WiFi.begin(ssid, password);
  // Important: disable Wi-Fi power-saving for stable streaming
  WiFi.setSleep(false);
  while (WiFi.status() != WL_CONNECTED) {
    delay(300);
    Serial.print(".");
  }
  Serial.println("\nWi-Fi connected");
  Serial.print("IP address: ");
  Serial.println(WiFi.localIP());

  // Start HTTP server and bind stream handler
  server.on("/stream", HTTP_GET, handleStream);
  server.begin();
  Serial.println("MJPEG stream ready at: ");
  Serial.print("  http://");
  Serial.print(WiFi.localIP());
  Serial.println("/stream");
}

void loop() {
  server.handleClient();
}
